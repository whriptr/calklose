#include <iostream>
#include <string>
//#include <iomanip>
#include <fstream>
#include <vector>
using namespace std;

int main()
{
	ifstream worldSeriesWinnersTxt;
	worldSeriesWinnersTxt.open("WorldSeriesWinners.txt");
	vector<string> winners;  // array to hold the teams that won the world series
	string teamName;  // variable for getline 
	string teamChoice;  // the team to search for
	int total = 0;  // number of world series wins

	if (!worldSeriesWinnersTxt)  // is the file open?
		cout << "error file not open\n";
	else  // if the file is open run the program
	{
		cout << "creating array...\n";
		while (!worldSeriesWinnersTxt.eof())  // read the text file and create the array
		{
			//winners.push_back(getline(worldSeriesWinnersTxt));  // this was the line I could not get to work
			getline(worldSeriesWinnersTxt, teamName);
			winners.push_back(teamName);
		}
		worldSeriesWinnersTxt.close();
		cout << "done\n";

		cout << "enter a team name to search for ";  // get a string to search for
		getline(cin, teamChoice);
		for (string val : winners)  // search the array for the string
		{
			if (teamChoice == val)  //  if the string is found, add one to the total
			{
				total++;
			}
		}
		cout << teamChoice << " have won " << total << " world series" << endl;  // print the results of the search
	}

	cout << endl << "press enter to close the program";
	cin.get();
	return 0;
}
